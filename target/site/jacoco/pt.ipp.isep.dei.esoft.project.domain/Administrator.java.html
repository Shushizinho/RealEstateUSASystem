    /**
     * Checks if the given object is equal to this Administrador object
     * @param obj the object to be compared with this Administrador object
     * @return true if the given object is equal to this Administrador object, false otherwise
     */
    public boolean equals(Object obj) {
        if (obj == this) {
            return true;
        }
        if (!(obj instanceof Administrator)) {
            return false;
        }
        Administrator other = (Administrator) obj;
        return Objects.equals(getEmployee(), other.getEmployee());
    }

    /**
     * Returns a hash code value for the Administrador object.
     * The hash code is based on the values of the employee field.
     *
     * @return the hash code value for this Administrador object
     */
    public int hashCode() {
        return Objects.hash(getEmployee());
    }

    /**
     *Returns a clone of this Administrador. The clone has the same employee as this Administrador.
     * @return a clone of this Administrador
     */
    public Administrator clone() {
        return new Administrator(this.getEmployee());
    }
